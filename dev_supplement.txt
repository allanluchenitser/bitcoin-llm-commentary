notes for local development

# run a workspace
npm run <script> -w <workspace-name-or-path>
npm run <script> -workspace <workspace-name-or-path>

# run all workspaces
npm run <script> -ws
npm run <script> --workspaces

# run containers, dependencies only for local dev
npm run base:up
npm run base:down

# run containers for everything
npm run apps:up
npm run apps:down
